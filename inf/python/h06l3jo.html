Циклические алгоритмы. Цикл while. Задачи

«B»: Напишите программу, которая получает два целых числа и находит их произведение, не используя операцию умножения. Учтите, что числа могут быть отрицательными.

print("Введите два числа:")
a, b = map(int, input().split())

multiply = 0

if (a >= 0) and (b >= 0): num1, num2 = a, b
elif (a >= 0) and (b <= 0): num1, num2 = a, -b
elif (a <= 0) and (b >= 0): num1, num2 = -a, b
else: num1, num2 = -a, -b

if num1 > num2: num1, num2 = num2, num1

while num2 > 0:
    multiply += num1
    num2 -= 1
    
if a >= 0 and b >= 0:
    print(a, "*", b, "=", multiply, sep="")
elif a >= 0 and b <= 0:
    print(a, "*(", b, ")=", -multiply, sep="")
elif a <= 0 and b >= 0:
    print("(", a, ")*", b, "=", -multiply, sep="")
else:
    print("(", a, ")*(", b, ")=", multiply, sep="")
